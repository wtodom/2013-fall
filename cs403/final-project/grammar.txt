program: functionDef optStatements

functionDef: TO VARIABLE optSequence COLON block

optSequence: sequence
	   | *empty*

sequence: primary
	| primary AND primary
	| primary commaChain COMMA AND primary

commaChain: COMMA primary commaChain
		  | *empty*

block: optStatements PERIOD

optStatements: statements
			 | *empty*

statements: statement optStatements

statement: expr SEMICOLON
		 | ifStatement
		 | whileStatement
		 | TODO
		 | assignment

ifStatement: IF primary IS primary COMMA block optOtherwise

whileStatement: WHILE primary IS primary COMMA block

optOtherwise: OTHERWISE block
			| OTHERWISE ifStatement
			| *empty*

assignment: SET variable TO expr

expr: OPEN_PARENTHESIS primary operator primary CLOSE_PARENTHESIS

primary: VARIABLE
	   | INTEGER
	   | STRING
	   | TRUE
	   | FLASE
	   | NOTHING (NULL)
	   | functionCall

functionCall: IDENTIFIER OPEN_PARENTHESIS optSequence CLOSE_PARENTHESIS

operator: PLUS
		| MINUS
		| DIVIDE
		| MULTIPLY
		| DOUBLE_EQUALS
		| NOT_EQUAL
		| LESS_THAN
		| GREATER_THAN
		| LESS_THAN_EQUAL
		| GREATER_THAN_EQUAL
