ifStatement: IF primary IS primary COMMA block optOtherwise

whileStatement: WHILE primary IS primary COMMA block

block: optStatements PERIOD

optOtherwise: OTHERWISE block
			| OTHERWISE ifStatement
			| *empty*

optStatements: statements
			 | *empty*

statements: statement optStatements

statement: expr SEMICOLON
		 | ifStatement
		 | whileStatement
		 | TODO

primary: VARIABLE
	   | INTEGER
	   | STRING
	   | TRUE
	   | FALSE
	   | NOTHING (NULL)
	   | expr

optList: list
	   | *empty*

list: primary
	| primary AND primary
	| primary commaChain COMMA AND primary

commaChain: COMMA primary commaChain
		  | *empty*

functionDef: TO VARIABLE optList COLON

expr: OPEN_PARENTHESIS primary operator primary CLOSE_PARENTHESIS

operator: PLUS
		| MINUS
		| DIVIDE
		| MULTIPLY
		| DOUBLE_EQUALS
		| NOT_EQUAL
		| LESS_THAN
		| GREATER_THAN
		| LESS_THAN_EQUAL
		| GREATER_THAN_EQUAL

functionCall: IDENTIFIER OPEN_PARENTHESIS optList CLOSE_PARENTHESIS
